        this.objectAttributes = objectAttributes;
    }

        return mergeRequest;
    }

        this.mergeRequest = mergeRequest;
    }

    @Override
    public boolean equals(Object o) {
        if (this == o) {
            return true;

        if (o == null || getClass() != o.getClass()) {

        return new EqualsBuilder()
